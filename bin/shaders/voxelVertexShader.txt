#version 330

in vec3 position;
in float textureCoordinates;

out vec2 textureCoords;
out float lightLevel;
out float layerF;

uniform mat4 transformationMatrix;

void main(void){
	//vec4 worldPosition = transformationMatrix * vec4(position,1.0);
	//gl_Position = projectionMatrix * viewMatrix * worldPosition;
	vec3 posadj = position - 0.5f;
	gl_Position = transformationMatrix * vec4(posadj,1.0);
	
	float x = ((int(textureCoordinates) >> 1) & 0x1);
	float y = ((int(textureCoordinates) & 0x1));
	
	textureCoords = vec2(x,y);
	
	lightLevel = ((int(textureCoordinates) >> 2) & 0xF) + ((int(textureCoordinates) >> 6) & 0xF);
	
	if (lightLevel > 15){
		lightLevel = 15;
	}
	layerF = (int(textureCoordinates) >> 10) & 0x2FF;
}
